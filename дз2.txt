#1) персонажи игры "baldur's gate 2". Список
npc_List = [
    ('Аномен', 'Муж.', 'Человек', 'Воин-Клирик', 'Законопослушный-нейтральный'),
    ('Aэри', 'Жен.', 'Эльф', 'Клирик-маг', 'Законопослушный-добрый'),
    ('Минск', 'Муж.', 'Человек', 'Следопыт', 'Хаотично-добрый'),
    ('Викония', 'Жен.', 'Дроу', 'Клирик', 'Нейтрально-злой')
]
#В данном случае список удобен своей краткостью. Однако в нем, в отличие от словаря, нет расшифровки каждого из элементов - это может привести 
#к неверному пониманию смысла элементов. Так же возможны дубли.

#Поиск по списку:
#найдем всех людей:
searchAttr = 'Человек'
for npc in npc_List:
    for prop in npc:
        if prop == searchAttr:
            print(npc[0])

#3) персонажи игры "baldur's gate 2". Множество   
npc_set = {
    ('Аномен', 'Муж.', 'Человек', 'Воин-Клирик', 'Законопослушный-нейтральный'),
    ('Аэри', 'Жен.', 'Эльф', 'Клирик-маг', 'Законопослушный-добрый'),
    ('Минск', 'Муж.', 'Человек', 'Следопыт', 'Хаотично-добрый'),
    ('Викония', 'Жен.', 'Дроу', 'Клирик', 'Нейтрально-злой')   
}
#Множество - этот способ похож на список, но он позволит исключить возможные дубли. 
#Однако, у элементов множества нет постоянного индекса, в отличие от списка. Возможно, поиск по множеству проходит быстрее чем по списку и словарю.

#найдем всех клириков (не важно, "чистых" или "мультиклассовых")
print('________________________')
searchAttr = 'клирик'
for npc in npc_set:
    for prop in npc:
        if searchAttr in prop.lower():
            print(npc[0], '-', prop)
